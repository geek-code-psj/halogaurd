```typescript
export const sendMessage = (type: string, payload: any) => {
  return new Promise((resolve, reject) => {
    chrome.tabs.query({ active: true }, (tabs) => {
      chrome.tabs.sendMessage(tabs[0].id, { type, payload }, (response) => {
        if (chrome.runtime.lastError) reject(chrome.runtime.lastError);
        resolve(response);
      });
    });
  });
};

export const listenForMessages = (callback: (msg: ExtensionMessage) => void) => {
  chrome.runtime.onMessage.addListener(callback);
};
```

**utils/api.ts**:
```typescript
export const analyzeContent = async (
  content: string,
  apiUrl: string
): Promise<AnalysisResult> => {
  try {
    // Try HTTP first
    const response = await fetch(`${apiUrl}/api/v1/analyze`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ content, model: 'gpt-4' })
    });
    
    if (!response.ok) throw new Error('API error');
    return await response.json();
  } catch (e) {
    // Fallback to WebSocket if needed
    return analyzeViaWebSocket(content, apiUrl);
  }
};
```

```typescript
chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
  if (request.type === 'ANALYZE_REQUEST') {
    // Get selected text or latest AI response
    const selectedText = window.getSelection().toString() ||
                         document.querySelector('[data-ai-response]')?.textContent;
    
    analyzeAndDisplay(selectedText, request.payload.apiUrl)
      .then(result => sendResponse({ status: 'success', result }))
      .catch(err => sendResponse({ status: 'error', error: err.message }));
  }
  return true; // Keep channel open for async response
});
```

## Notes

### Known Challenges
1. **CSP (Content Security Policy)**: ChatGPT/Claude have strict CSP - must inject carefully
   - Solution: Use message passing, avoid inline scripts
2. **DOM Injection**: Need to find correct elements for ChatGPT/Claude different interfaces
   - Solution: Add platform-specific selectors
3. **Rate Limiting**: Backend limits requests/sec
   - Solution: Batch requests, add debouncing (max 1 analysis/5s)
